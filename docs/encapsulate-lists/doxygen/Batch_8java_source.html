74 public Batch(Config config, ArchiveQuery aq, KafkaQuery kq) {
75 this.config = config;
76 this.runQueueArray = new LinkedList<>();
77 this.batchList = new LinkedList<>();
78
79 for (int i = 0; i < config.batchConfig.numPartitions; i++) {
80 this.runQueueArray.add(new BatchTaskQueue());
81 }
82
83 this.archiveQuery = aq;

87 public LinkedList<LinkedList<BatchSlice>> processRange(Offset start, Offset end) {
88 LOGGER.debug("processRange");
89
90 final List<BatchSlice> slices = new LinkedList<>();
91 if (config.isArchiveEnabled) {
92 slices.addAll(new ArchiveBatchFactory(this.archiveQuery).fromRange(start, end));
93 }
94
95 if (config.isKafkaEnabled) {
96 slices.addAll(new KafkaBatchFactory(this.kafkaQuery).fromRange(start, end));
97 }
98
99 if (!slices.isEmpty()) {
100 this.addSlice(slices);
101 }
102
103 return this.getBatch();
104 }

106 public void addSlice(List<BatchSlice> sliceCollection) {
107 while (!sliceCollection.isEmpty()) {
108 BatchSlice longestObject = null;
109
110 // find the longest object
111 for (BatchSlice objectMetadata : sliceCollection) {
112 if (longestObject == null) {
113 longestObject = objectMetadata;
114 }
115 else {
116 if (longestObject.getSize() > objectMetadata.getSize()) {
117 longestObject = objectMetadata;
118 }
119 }
120 }
121
122 // object found
123 if (longestObject != null) {
124 sliceCollection.remove(longestObject);
125
126 // find shortest queue
127 BatchTaskQueue shortestQueue = null;
128 for (final BatchTaskQueue btq : runQueueArray) {
129 if (shortestQueue == null) {
130 shortestQueue = btq;
131 }
132 else {
133 if (shortestQueue.getQueueTime() > btq.getQueueTime()) {
134 shortestQueue = btq;
135 }
136 }
137 }
138 if (shortestQueue != null) {
139 shortestQueue.add(longestObject);
140 }
141 else {
142 throw new RuntimeException("shortestQueue was null");
143 }
144
145 }
146 }
147 }
148
149 public LinkedList<LinkedList<BatchSlice>> getBatch() {
150 batchList.clear(); // clear on getBatch?
151 for (final BatchTaskQueue btq : runQueueArray) {
152 batchList.add(btq.getQueue());
153 }
154 LOGGER.debug("getBatch: <{}>", batchList);
155 return batchList;
156 }
